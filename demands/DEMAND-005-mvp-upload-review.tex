\section{具体任务}

\subsection{后端}

\begin{enumerate}[nosep]
  \item 用户认证：JWT 登录/注册，种子数据创建管理员账号
  \item Skill 上传接口：POST /api/skills/upload，接收 zip 文件，存储到 MinIO
  \item 三层扫描服务：
    \begin{itemize}[nosep]
      \item Semgrep：解压 zip 后对代码文件执行静态扫描
      \item ClamAV：对 zip 包执行恶意文件扫描
      \item qwen-flash：提取 zip 内的关键文件（如 manifest/yaml/config），调用 qwen-turbo API 做语义分析
    \end{itemize}
  \item 审批接口：GET /api/reviews（列表）、POST /api/reviews/\{id\}/approve、POST /api/reviews/\{id\}/reject
  \item Skill CRUD：GET /api/skills（列表）、GET /api/skills/\{id\}（详情含扫描报告）
  \item 种子数据：虚设 3-5 个 Skill + 1 个管理员账号（admin/admin123）
\end{enumerate}

\subsection{前端}

\begin{enumerate}[nosep]
  \item 登录页：用户名+密码，黑白灰风格
  \item Skill 上传页：拖拽或点击上传 zip，上传后自动跳转到审批状态
  \item 审批工作台（核心页面）：
    \begin{itemize}[nosep]
      \item 待审批列表（表格形式）
      \item 点击进入详情：展示三层扫描结果
      \item 每层扫描结果：通过/不通过 + 告警详情（展开查看）
      \item 操作按钮：通过 / 驳回（+ 备注）
    \end{itemize}
  \item Skill 列表页：已上架的 Skill 卡片展示
\end{enumerate}

\subsection{环境}

\begin{enumerate}[nosep]
  \item Docker Compose 启动所有服务
  \item 后端绑定 0.0.0.0:8000（局域网可访问）
  \item 前端绑定 0.0.0.0:5173（局域网可访问）
  \item ClamAV 安装到后端 Docker 镜像或作为独立服务
\end{enumerate}
